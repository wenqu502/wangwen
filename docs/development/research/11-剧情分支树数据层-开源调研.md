### 2.3 树形结构 vs DAG

**当前设计**：树形结构（单父节点）
- 优点：查询简单，UI 展示直观
- 缺点：无法表达"合并"场景（两条支线汇聚）

**扩展预留**：DAG 结构（多父节点）

## 三、树查询算法

### 3.1 获取子树

```typescript
async function getSubTree(rootId: string): Promise<PlotNode[]> {
  const result: PlotNode[] = []
  const queue = [rootId]
  
  while (queue.length > 0) {
    const parentId = queue.shift()!
    const children = await db.plotNodes
      .where('parentId')
      .equals(parentId)
      .sortBy('order')
    
    result.push(...children)
    queue.push(...children.map(c => c.id))
  }
  
  return result
}
```

### 3.2 获取祖先链

```typescript
async function getAncestorChain(nodeId: string): Promise<PlotNode[]> {
  const chain: PlotNode[] = []
  let current = await db.plotNodes.get(nodeId)
  
  while (current?.parentId) {
    current = await db.plotNodes.get(current.parentId)
    if (current) chain.unshift(current)
  }
  
  return chain
}
```

### 3.3 路径查询（两点间）

```typescript
async function findPath(startId: string, endId: string): Promise<PlotNode[] | null> {
  // BFS
  const visited = new Set<string>()
  const queue: Array<{ id: string; path: string[] }> = [{ id: startId, path: [startId] }]
  
  while (queue.length > 0) {
    const { id, path } = queue.shift()!
    
    if (id === endId) {
      return await Promise.all(path.map(pid => db.plotNodes.get(pid)))
    }
    
    if (visited.has(id)) continue
    visited.add(id)
    
    const children = await db.plotNodes.where('parentId').equals(id).toArray()
    for (const child of children) {
      queue.push({ id: child.id, path: [...path, child.id] })
    }
  }
  
  return null
}
```

```typescript
async function detectForeshadowings(nodeId: string): Promise<Foreshadowing[]> {
  const node = await db.plotNodes.get(nodeId)
  if (!node?.content) return []
  
  const prompt = `
请分析以下章节内容，识别潜在的伏笔：

章节标题：${node.title}
章节内容：${node.content}

伏笔是指：
- 未解释的异常事件
- 突然出现的物品/人物
- 角色说出的意味深长的话
- 被刻意忽略的细节

请以 JSON 数组格式输出，每个伏笔包含：
- description: 伏笔描述
- type: 类型（item/character/dialogue/detail）
`
  
  const result = await aiGenerateObject(prompt, ForeshadowingArraySchema)
  
  return result.map(f => ({
    id: generateId(),
    workId: node.workId,
    description: f.description,
    sourceNodeId: nodeId,
    status: 'planted',
    reminderThreshold: 10, // 默认10章
    createdAt: Date.now(),
  }))
}
```

### 4.3 伏笔提醒检查

```typescript
async function checkForeshadowingReminders(workId: string) {
  const foreshadowings = await db.foreshadowings
    .where('workId')
    .equals(workId)
    .and(f => f.status === 'planted')
    .toArray()
  
  const currentNodeCount = await db.plotNodes
    .where('workId')
    .equals(workId)
    .count()
  
  const reminders: Foreshadowing[] = []
  
  for (const fs of foreshadowings) {
    const sourceNode = await db.plotNodes.get(fs.sourceNodeId)
    if (!sourceNode) continue
    
    const nodesSince = currentNodeCount - sourceNode.order
    
    if (nodesSince > fs.reminderThreshold) {
      const shouldRemind = !fs.lastRemindedAt || 
        (Date.now() - fs.lastRemindedAt) > 1000 * 60 * 60 * 24 // 每天最多提醒一次
      
      if (shouldRemind) {
        reminders.push(fs)
        await db.foreshadowings.update(fs.id, { lastRemindedAt: Date.now() })
      }
    }
  }
  
  return reminders
}
```

## 五、Zustand Store 设计

```typescript
// src/modules/plot/store.ts

interface PlotState {
  nodes: PlotNode[]
  edges: PlotEdge[]
  selectedNodeId: string | null
  foreshadowings: Foreshadowing[]
  isLoading: boolean
  
  // Actions
  loadPlotTree: (workId: string) => Promise<void>
  createNode: (workId: string, data: PlotNodeForm) => Promise<string>
  updateNode: (id: string, data: Partial<PlotNode>) => Promise<void>
  deleteNode: (id: string) => Promise<void>
  moveNode: (id: string, parentId: string | null, order: number) => Promise<void>
  setNodePosition: (id: string, position: { x: number; y: number }) => void
  selectNode: (id: string | null) => void
  checkForeshadowings: (workId: string) => Promise<void>
}

export const usePlotStore = create<PlotState>()(
  immer((set, get) => ({
    nodes: [],
    edges: [],
    selectedNodeId: null,
    foreshadowings: [],
    isLoading: false,
    
    async loadPlotTree(workId) {
      set({ isLoading: true })
      const [nodes, edges, fss] = await Promise.all([
        db.plotNodes.where('workId').equals(workId).toArray(),
        db.plotEdges.where('workId').equals(workId).toArray(),
        db.foreshadowings.where('workId').equals(workId).toArray(),
      ])
      set({ nodes, edges, foreshadowings: fss, isLoading: false })
    },
    
    async createNode(workId, data) {
      const node: PlotNode = {
        id: generateId(),
        workId,
        ...data,
        createdAt: Date.now(),
        updatedAt: Date.now(),
      }
      
      await db.transaction('rw', [db.plotNodes, db.plotEdges], async () => {
        await db.plotNodes.add(node)
        if (data.parentId) {
          await db.plotEdges.add({
            id: generateId(),
            workId,
            sourceId: data.parentId,
            targetId: node.id,
            type: data.type === 'branch' ? 'branch' : 'main',
          })
        }
      })
      
      set(state => { state.nodes.push(node) })
      return node.id
    },
    
    async moveNode(id, parentId, order) {
      await db.plotNodes.update(id, { parentId, order, updatedAt: Date.now() })
      set(state => {
        const node = state.nodes.find(n => n.id === id)
        if (node) {
          node.parentId = parentId
          node.order = order
          node.updatedAt = Date.now()
        }
      })
    },
    
    setNodePosition(id, position) {
      set(state => {
        const node = state.nodes.find(n => n.id === id)
        if (node) node.position = position
      })
      // 防抖保存到数据库
      debouncedSavePosition(id, position)
    },
  }))
)
```
