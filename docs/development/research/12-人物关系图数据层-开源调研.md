### 2.3 图数据结构特点

- **无向图为主**：大部分关系是双向的（如"挚友"）
- **有向图为辅**：部分关系有方向（如"爱慕"可能是单向的）
- **边属性丰富**：类型、强度、隐藏状态等
- **动态变化**：随剧情推进，关系会变化

### 3.2 最短路径查询（两个角色间）

```typescript
async function findShortestPath(sourceId: string, targetId: string): Promise<string[] | null> {
  if (sourceId === targetId) return [sourceId]
  
  const visited = new Set<string>()
  const queue: Array<{ id: string; path: string[] }> = [{ id: sourceId, path: [sourceId] }]
  
  while (queue.length > 0) {
    const { id, path } = queue.shift()!
    
    if (id === targetId) return path
    if (visited.has(id)) continue
    visited.add(id)
    
    const relations = await getCharacterRelations(id)
    for (const rel of relations) {
      const nextId = rel.sourceId === id ? rel.targetId : rel.sourceId
      if (!visited.has(nextId)) {
        queue.push({ id: nextId, path: [...path, nextId] })
      }
    }
  }
  
  return null
}
```

### 3.3 关系网络统计

```typescript
async function getRelationStats(workId: string) {
  const relations = await db.relations.where('workId').equals(workId).toArray()
  
  return {
    total: relations.length,
    byCategory: {
      emotion: relations.filter(r => r.category === 'emotion').length,
      blood: relations.filter(r => r.category === 'blood').length,
      interest: relations.filter(r => r.category === 'interest').length,
      hidden: relations.filter(r => r.category === 'hidden').length,
    },
    byType: relations.reduce((acc, r) => {
      acc[r.type] = (acc[r.type] || 0) + 1
      return acc
    }, {} as Record<string, number>),
    averageStrength: relations.reduce((sum, r) => sum + r.strength, 0) / relations.length,
  }
}
```

## 四、AI 自动推断

### 4.1 推断流程

```typescript
async function inferRelations(workId: string): Promise Promise<RelationEdge[]> {
  const characters = await db.characters.where('workId').equals(workId).toArray()
  const plotNodes = await db.plotNodes.where('workId').equals(workId).toArray()
  
  // 1. 构建角色描述
  const characterDescriptions = characters.map(c => ({
    id: c.id,
    name: c.name,
    summary: `${c.name}: ${c.background}。性格：${c.personality.surface}。`,
  }))
  
  // 2. 构建剧情摘要
  const plotSummary = plotNodes.map(n => `${n.title}: ${n.summary}`).join('\n')
  
  // 3. AI 推断
  const prompt = `
根据以下角色档案和剧情，推断人物之间的关系。

角色：
${characterDescriptions.map(c => c.summary).join('\n')}

剧情：
${plotSummary}

请以 JSON 数组输出，每个关系包含：
- sourceId: 源角色 ID
- targetId: 目标角色 ID
- type: 关系类型
- category: 类别（emotion/blood/interest/hidden）
- description: 关系描述
- directed: 是否有向
- strength: 强度（1-10）
`
  
  const inferred = await aiGenerateObject(prompt, RelationArraySchema)
  
  // 4. 过滤已存在的关系
  const existing = await db.relations.where('workId').equals(workId).toArray()
  const existingPairs = new Set(existing.map(r => `${r.sourceId}-${r.targetId}`))
  
  return inferred.filter((r: RelationEdge) => 
    !existingPairs.has(`${r.sourceId}-${r.targetId}`)
  )
}
```

### 4.2 矛盾检测

```typescript
async function detectRelationConflicts(workId: string): Promise Promise<Conflict[]> {
  const relations = await db.relations.where('workId').equals(workId).toArray()
  const plotNodes = await db.plotNodes.where('workId').equals(workId).toArray()
  
  const conflicts: Conflict[] = []
  
  for (const rel of relations) {
    const [source, target] = await Promise.all([
      db.characters.get(rel.sourceId),
      db.characters.get(rel.targetId),
    ])
    
    if (!source || !target) continue
    
    // AI 检测矛盾
    const prompt = `
检查以下关系是否与角色设定或剧情矛盾：

关系：${source.name} → ${target.name}: ${rel.type}（${rel.description}）

${source.name} 设定：${source.background}
${target.name} 设定：${target.background}

是否有矛盾？如有，请说明。
`
    const analysis = await aiAnalyze(prompt)
    if (analysis.hasConflict) {
      conflicts.push({
        relationId: rel.id,
        description: analysis.description,
      })
    }
  }
  
  return conflicts
}
```

## 五、Zustand Store 设计

```typescript
// src/modules/relation/store.ts

interface RelationState {
  edges: RelationEdge[]
  relationTypes: RelationType[]
  selectedEdgeId: string | null
  isLoading: boolean
  
  loadRelations: (workId: string) => Promise<void>
  createRelation: (data: Omit Omit<RelationEdge, 'id' | 'createdAt' | 'updatedAt'>) => Promise<string>
  updateRelation: (id: string, data: Partial Partial<RelationEdge>) => Promise<void>
  deleteRelation: (id: string) => Promise<void>
  inferRelations: (workId: string) => Promise<void>
  addRelationType: (type: Omit Omit<RelationType, 'id'>) => Promise<string>
}

export const useRelationStore = create create<RelationState>()(
  immer((set, get) => ({
    edges: [],
    relationTypes: [],
    selectedEdgeId: null,
    isLoading: false,
    
    async loadRelations(workId) {
      set({ isLoading: true })
      const [edges, types] = await Promise.all([
        db.relations.where('workId').equals(workId).toArray(),
        db.relationTypes.where('workId').equals(workId).toArray(),
      ])
      set({ edges, relationTypes: types, isLoading: false })
    },
    
    async createRelation(data) {
      const edge: RelationEdge = {
        id: generateId(),
        ...data,
        createdAt: Date.now(),
        updatedAt: Date.now(),
      }
      await db.relations.add(edge)
      set(state => { state.edges.push(edge) })
      return edge.id
    },
    
    async inferRelations(workId) {
      set({ isLoading: true })
      const inferred = await inferRelations(workId)
      
      if (inferred.length > 0) {
        await db.relations.bulkAdd(inferred)
        set(state => {
          state.edges.push(...inferred)
        })
      }
      
      set({ isLoading: false })
    },
  }))
)
```
