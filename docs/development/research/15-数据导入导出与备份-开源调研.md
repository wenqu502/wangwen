### 1.1 PRD 原文需求摘录

```typescript
// src/utils/export.ts

export interface WorkExport {
  version: string
  exportedAt: string
  work: Work
  characters: Character[]
  plotNodes: PlotNode[]
  plotEdges: PlotEdge[]
  relations: RelationEdge[]
  relationTypes: RelationType[]
  systems: WorkSystem[]
  ideas: Idea[]
  foreshadowings: Foreshadowing[]
  chatHistory?: ChatMessage[]
}

export async function exportWork(workId: string): Promise Promise<WorkExport> {
  const work = await db.works.get(workId)
  if (!work) throw new Error('作品不存在')

  const [
    characters,
    plotNodes,
    plotEdges,
    relations,
    relationTypes,
    systems,
    ideas,
    foreshadowings,
  ] = await Promise.all([
    db.characters.where('workId').equals(workId).toArray(),
    db.plotNodes.where('workId').equals(workId).toArray(),
    db.plotEdges.where('workId').equals(workId).toArray(),
    db.relations.where('workId').equals(workId).toArray(),
    db.relationTypes.where('workId').equals(workId).toArray(),
    db.systems.where('workId').equals(workId).toArray(),
    db.ideas.where('workId').equals(workId).toArray(),
    db.foreshadowings.where('workId').equals(workId).toArray(),
  ])

  return {
    version: 'wangwen-export-v1',
    exportedAt: new Date().toISOString(),
    work,
    characters,
    plotNodes,
    plotEdges,
    relations,
    relationTypes,
    systems,
    ideas,
    foreshadowings,
  }
}

export function downloadJSON(data: unknown, filename: string) {
  const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = filename
  document.body.appendChild(a)
  a.click()
  document.body.removeChild(a)
  URL.revokeObjectURL(url)
}
```

### 2.2 图片导出（html2canvas）

```typescript
import html2canvas from 'html2canvas'

export async function exportToImage(
  element: HTMLElement,
  filename: string,
  options: { scale?: number; backgroundColor?: string } = {}
) {
  const canvas = await html2canvas(element, {
    scale: options.scale || 2,
    backgroundColor: options.backgroundColor || '#f5f7fa',
    logging: false,
    useCORS: true,
    allowTaint: true,
  })

  canvas.toBlob((blob) => {
    if (!blob) return
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = filename
    a.click()
    URL.revokeObjectURL(url)
  }, 'image/png')
}
```

**注意事项**：
- 跨域图片需要 `useCORS: true`
- 大数据量画布可能导致内存溢出
- 建议分块导出或降低 scale

### 2.3 PDF 导出（jsPDF）

```typescript
import jsPDF from 'jspdf'

export async function exportToPDF(
  element: HTMLElement,
  filename: string
) {
  const canvas = await html2canvas(element, { scale: 2 })
  const imgData = canvas.toDataURL('image/png')

  const pdf = new jsPDF({
    orientation: canvas.width > canvas.height ? 'landscape' : 'portrait',
    unit: 'px',
    format: [canvas.width, canvas.height],
  })

  pdf.addImage(imgData, 'PNG', 0, 0, canvas.width, canvas.height)
  pdf.save(filename)
}
```

**多页 PDF（长内容）**：
```typescript
export async function exportLongContentToPDF(
  elements: HTMLElement[],
  filename: string
) {
  const pdf = new jsPDF('p', 'mm', 'a4')
  const pageWidth = 210
  const pageHeight = 297

  for (let i = 0; i < elements.length; i++) {
    if (i > 0) pdf.addPage()

    const canvas = await html2canvas(elements[i], { scale: 2 })
    const imgData = canvas.toDataURL('image/png')

    const imgWidth = pageWidth
    const imgHeight = (canvas.height * imgWidth) / canvas.width

    pdf.addImage(imgData, 'PNG', 0, 0, imgWidth, Math.min(imgHeight, pageHeight))
  }

  pdf.save(filename)
}
```

## 三、数据导入方案

### 3.1 JSON 导入

```typescript
export async function importWork(file: File): Promise<string> {
  const text = await file.text()
  const data = JSON.parse(text) as WorkExport

  // 验证版本
  if (!data.version?.startsWith('wangwen-export')) {
    throw new Error('不支持的导出文件格式')
  }

  // 生成新 ID 避免冲突
  const idMap = new Map<string, string>()
  const newWorkId = generateId()

  const remapId = (oldId: string): string => {
    if (!idMap.has(oldId)) {
      idMap.set(oldId, generateId())
    }
    return idMap.get(oldId)!
  }

  await db.transaction(
    'rw',
    [db.works, db.characters, db.plotNodes, db.plotEdges, db.relations,
     db.relationTypes, db.systems, db.ideas, db.foreshadowings],
    async () => {
      // 导入作品
      await db.works.add({
        ...data.work,
        id: newWorkId,
        name: `${data.work.name} (导入)`,
        createdAt: Date.now(),
        updatedAt: Date.now(),
      })

      // 导入角色
      if (data.characters.length > 0) {
        await db.characters.bulkAdd(
          data.characters.map(c => ({
            ...c,
            id: remapId(c.id),
            workId: newWorkId,
          }))
        )
      }

      // 导入剧情节点（需要先建立 ID 映射）
      if (data.plotNodes.length > 0) {
        await db.plotNodes.bulkAdd(
          data.plotNodes.map(n => ({
            ...n,
            id: remapId(n.id),
            workId: newWorkId,
            parentId: n.parentId ? remapId(n.parentId) : null,
          }))
        )
      }

      // 导入关系边
      if (data.plotEdges.length > 0) {
        await db.plotEdges.bulkAdd(
          data.plotEdges.map(e => ({
            ...e,
            id: generateId(),
            workId: newWorkId,
            sourceId: remapId(e.sourceId),
            targetId: remapId(e.targetId),
          }))
        )
      }

      // ... 其他表
    }
  )

  return newWorkId
}
```

### 3.2 导入验证

```typescript
function validateImportData(data: unknown): data is WorkExport {
  if (!data || typeof data !== 'object') return false

  const d = data as Partial Partial<WorkExport>

  // 检查必填字段
  if (!d.version || !d.work || !Array.isArray(d.characters)) return false

  // 检查版本兼容性
  const version = d.version.replace('wangwen-export-v', '')
  if (parseInt(version) > 1) {
    console.warn(`导入文件版本 ${d.version} 高于当前支持版本，可能不兼容`)
  }

  return true
}
```

## 四、自动备份方案

### 4.1 备份策略

```typescript
// src/utils/auto-backup.ts

const DEFAULT_CONFIG = {
  interval: 1000 * 60 * 30, // 30分钟
  maxBackups: 10,
  includeChatHistory: false,
}

interface BackupRecord {
  id: string
  timestamp: number
  data: WorkExport[]
  size: number
}

class AutoBackup {
  private timer: ReturnTypeType<typeof setInterval> | null = null
  private config = DEFAULT_CONFIG

  start() {
    if (this.timer) return
    this.timer = setInterval(() => this.createBackup(), this.config.interval)
  }

  stop() {
    if (this.timer) {
      clearInterval(this.timer)
      this.timer = null
    }
  }

  async createBackup(): Promise<string> {
    const works = await db.works.toArray()
    const backupData = await Promise.all(
      works.map(w => exportWork(w.id))
    )

    const record: BackupRecord = {
      id: `backup_${Date.now()}`,
      timestamp: Date.now(),
      data: backupData,
      size: JSON.stringify(backupData).length,
    }

    await db.backups.add(record)
    await this.cleanOldBackups()

    return record.id
  }

  private async cleanOldBackups() {
    const oldBackups = await db.backups
      .orderBy('timestamp')
      .reverse()
      .offset(this.config.maxBackups)
      .toArray()

    if (oldBackups.length > 0) {
      await db.backups.bulkDelete(oldBackups.map(b => b.id))
    }
  }

  async restoreBackup(backupId: string): Promise<void> {
    const backup = await db.backups.get(backupId)
    if (!backup) throw new Error('备份不存在')

    for (const workData of backup.data) {
      // 生成新 ID 导入，避免覆盖现有数据
      await importWorkFromData(workData)
    }
  }

  async listBackups(): Promise Promise<BackupRecord[]> {
    return await db.backups
      .orderBy('timestamp')
      .reverse()
      .toArray()
  }
}

export const autoBackup = new AutoBackup()
```

```typescript
// src/utils/storage-monitor.ts

export async function getStorageStats(): Promise Promise<StorageQuota> {
  const estimate = await navigator.storage.estimate()
  const usage = estimate.usage || 0
  const quota = estimate.quota || 0

  // 计算各表占用
  const tableSizes: Record<string, number> = {}

  for (const tableName of ['works', 'characters', 'plotNodes', 'relations', 'systems', 'ideas', 'backups']) {
    const table = (db as any)[tableName]
    if (table) {
      const count = await table.count()
      tableSizes[tableName] = count
    }
  }

  return {
    usedMB: (usage / 1024 / 1024).toFixed(2),
    totalMB: (quota / 1024 / 1024).toFixed(2),
    percent: ((usage / quota) * 100).toFixed(1),
    tableSizes,
  }
}

// 定期监控
export function startStorageMonitoring() {
  setInterval(async () => {
    const stats = await getStorageStats()
    if (parseFloat(stats.percent) > 80) {
      // 触发预警
      eventBus.emit('storage:warning', stats)
    }
  }, 1000 * 60 * 30) // 每30分钟
}
```

## 五、数据迁移（Schema 升级）

### 5.1 Dexie 版本升级

```typescript
class WangWenDB extends Dexie {
  constructor() {
    super('wangwen-db')

    // v1: 初始版本
    this.version(1).stores({
      works: 'id, name, createdAt',
      characters: 'id, workId, name, [workId+name]',
      // ...
    })

    // v2: 新增角色状态字段
    this.version(2)
      .stores({
        characters: 'id, workId, name, status, [workId+name], [workId+status]',
      })
      .upgrade(async (tx) => {
        await tx.table('characters').toCollection().modify((char) => {
          char.status = char.status || 'alive'
        })
      })

    // v3: 新增作品标签
    this.version(3)
      .stores({
        works: 'id, name, *tags, createdAt',
      })
      .upgrade(async (tx) => {
        await tx.table('works').toCollection().modify((work) => {
          work.tags = work.tags || []
        })
      })
  }
}
```

```typescript
async function upgradeWithBackup(
  newVersion: number,
  upgradeFn: () => Promise<void>
): Promise<void> {
  // 1. 备份当前数据
  const backup = await autoBackup.createBackup()
  console.log(`升级前备份完成: ${backup}`)

  try {
    // 2. 执行升级
    await upgradeFn()
    console.log(`数据库升级到 v${newVersion} 成功`)
  } catch (err) {
    // 3. 升级失败，提示恢复
    console.error('升级失败:', err)
    throw new Error(`数据库升级失败，请从备份 ${backup} 恢复`)

}
```

## 六、Zustand Store 设计

```typescript
// src/stores/export-store.ts

interface ExportState {
  isExporting: boolean
  exportProgress: number
  lastBackupAt: number | null

  exportWork: (workId: string) => Promise<void>
  importWork: (file: File) => Promise<string>
  exportToImage: (element: HTMLElement, filename: string) => Promise<void>
  exportToPDF: (element: HTMLElement, filename: string) => Promise<void>
  createBackup: () => Promise<string>
  restoreBackup: (backupId: string) => Promise<void>
}

export const useExportStore = create create<ExportState>()(
  immer((set) => ({
    isExporting: false,
    exportProgress: 0,
    lastBackupAt: null,

    async exportWork(workId) {
      set({ isExporting: true, exportProgress: 0 })
      try {
        const data = await exportWork(workId)
        set({ exportProgress: 50 })
        downloadJSON(data, `wangwen-${data.work.name}-${Date.now()}.json`)
        set({ exportProgress: 100 })
      } finally {
        setTimeout(() => set({ isExporting: false, exportProgress: 0 }), 500)
      }
    },

    async importWork(file) {
      set({ isExporting: true })
      try {
        const workId = await importWork(file)
        return workId
      } finally {
        set({ isExporting: false })
      }
    },

    async createBackup() {
      const backupId = await autoBackup.createBackup()
      set({ lastBackupAt: Date.now() })
      return backupId
    },
  }))
)
```
