```typescript
this.version(1).stores({
  systems: 'id, workId, name, [workId+name]',
```

体系数据作为 JSON 嵌套存储在 `systems` 表中，因为：
- 体系内部结构固定（分支 → 等级）
- 查询以整个体系为单位
- 避免多表 JOIN 的复杂性

## 三、规则引擎设计

### 3.1 规则表示

```typescript
// 规则条件表达式（简化版）
interface RuleCondition {
  field: string      // 如 "character.level"
  operator: 'eq' | 'gt' | 'lt' | 'gte' | 'lte' | 'in'
  value: unknown
}

interface SystemRule {
  id: string
  description: string
  conditions: RuleCondition[]
  severity: 'hard' | 'soft' | 'suggestion'
  errorMessage: string
}
```

### 3.2 规则执行

```typescript
function evaluateRule(rule: SystemRule, context: Record<string, unknown>): boolean {
  return rule.conditions.every(condition => {
    const value = getValueByPath(context, condition.field)
    switch (condition.operator) {
      case 'eq': return value === condition.value
      case 'gt': return (value as number) > (condition.value as number)
      case 'lt': return (value as number) < (condition.value as number)
      case 'gte': return (value as number) >= (condition.value as number)
      case 'lte': return (value as number) <= (condition.value as number)
      case 'in': return (condition.value as unknown[]).includes(value)
      default: return false
    }
  })
}
```

### 3.3 AI 规则冲突检测

```typescript
async function checkRuleConflicts(workId: string): Promise Promise<Conflict[]> {
  const systems = await db.systems.where('workId').equals(workId).toArray()
  const plotNodes = await db.plotNodes.where('workId').equals(workId).toArray()
  const characters = await db.characters.where('workId').equals(workId).toArray()
  
  const conflicts: Conflict[] = []
  
  for (const node of plotNodes) {
    if (!node.content) continue
    
    for (const system of systems) {
      const prompt = `
检查以下剧情内容是否违反体系规则。

体系名称：${system.name}
体系规则：
${system.rules.map(r => `- ${r.description}（${r.severity}）`).join('\n')}

剧情内容：
${node.content}

如有违反，请输出：
- 违反的规则
- 冲突描述
- 建议修改
`
      const analysis = await aiAnalyze(prompt)
      if (analysis.hasConflict) {
        conflicts.push({
          nodeId: node.id,
          systemId: system.id,
          ruleId: analysis.ruleId,
          description: analysis.description,
        })
      }
    }
  }
  
  return conflicts
}
```

## 四、Zustand Store 设计

```typescript
// src/modules/system/store.ts

interface SystemState {
  systems: WorkSystem[]
  selectedSystemId: string | null
  isLoading: boolean
  
  loadSystems: (workId: string) => Promise<void>
  createSystem: (workId: string, data: SystemForm) => Promise<string>
  updateSystem: (id: string, data: Partial Partial<WorkSystem>) => Promise<void>
  deleteSystem: (id: string) => Promise<void>
  addBranch: (systemId: string, branch: SystemBranch) => Promise<void>
  addLevel: (systemId: string, branchId: string, level: SystemLevel) => Promise<void>
  addRule: (systemId: string, rule: SystemRule) => Promise<void>
  checkConflicts: (workId: string) => Promise Promise<Conflict[]>
}

export const useSystemStore = create<SystemState>()(
  immer((set, get) => ({
    systems: [],
    selectedSystemId: null,
    isLoading: false,
    
    async loadSystems(workId) {
      set({ isLoading: true })
      const systems = await db.systems.where('workId').equals(workId).toArray()
      set({ systems, isLoading: false })
    },
    
    async createSystem(workId, data) {
      const system: WorkSystem = {
        id: generateId(),
        workId,
        ...data,
        branches: [],
        rules: [],
        createdAt: Date.now(),
        updatedAt: Date.now(),
      }
      await db.systems.add(system)
      set(state => { state.systems.push(system) })
      return system.id
    },
    
    async addBranch(systemId, branch) {
      await db.systems.update(systemId, (system) => {
        system.branches.push(branch)
        system.updatedAt = Date.now()
      })
      set(state => {
        const system = state.systems.find(s => s.id === systemId)
        if (system) {
          system.branches.push(branch)
          system.updatedAt = Date.now()
        }
      })
    },
  }))
)
```
