### 3.1 提取流程

async function extractIdeasFromChat(workId: string, chatMessage: string): Promise<Idea[]> {
  const prompt = `
请从以下对话内容中提取灵感便签。

对话内容：
${chatMessage}

灵感是指：
- 突然想到的好点子
- 剧情转折的创意
- 角色发展的新方向
- 世界观的补充设定
- 经典台词或对话

请以 JSON 数组输出，每个灵感包含：
- title: 灵感标题（10字以内）
- content: 灵感内容（100字以内）
- category: 类别（plot/character/world/dialogue/other）
- tags: 标签数组

只提取有价值的灵感，不要过度提取。
`
  
  const extracted = await aiGenerateObject(prompt, IdeaArraySchema)
  
  return extracted.map((item: IdeaForm) => ({
    id: generateId(),
    workId,
    ...item,
    status: 'pending',
    source: 'ai-extract',
    characterIds: [],
    plotNodeIds: [],
    createdAt: Date.now(),
    updatedAt: Date.now(),
  }))
}

### 3.2 使用建议

```typescript
async function getIdeaRecommendations(workId: string, currentNodeId?: string): Promise<Idea[]> {
  const ideas = await db.ideas
    .where('workId')
    .equals(workId)
    .and(i => i.status === 'pending')
    .toArray()
  
  if (!currentNodeId) {
    // 按创建时间倒序返回
    return ideas.sort((a, b) => b.createdAt - a.createdAt).slice(0, 10)
  }
  
  const currentNode = await db.plotNodes.get(currentNodeId)
  if (!currentNode) return ideas.slice(0, 10)
  
  // AI 计算相关性
  const prompt = `
请对以下灵感便签与当前剧情节点的相关性进行评分（1-10）。

当前剧情节点：${currentNode.title}
${currentNode.summary}

灵感便签：
${ideas.map((i, idx) => `${idx + 1}. ${i.title}: ${i.content}`).join('\n')}

请输出每个灵感的相关性评分（JSON数组）。
`
  
  const scores = await aiGenerateObject(prompt, ScoreArraySchema)
  
  return ideas
    .map((idea, idx) => ({ idea, score: scores[idx] || 0 }))
    .sort((a, b) => b.score - a.score)
    .slice(0, 10)
    .map(({ idea }) => idea)
}
```

## 四、Zustand Store 设计

```typescript
// src/modules/idea/store.ts

interface IdeaState {
  ideas: Idea[]
  selectedIdeaId: string | null
  filter: IdeaFilter
  isLoading: boolean
  
  loadIdeas: (workId: string) => Promise<void>
  createIdea: (workId: string, data: IdeaForm) => Promise<string>
  updateIdea: (id: string, data: Partial<Idea>) => Promise<void>
  deleteIdea: (id: string) => Promise<void>
  markAsUsed: (id: string, nodeId: string) => Promise<void>
  archiveIdea: (id: string) => Promise<void>
  extractFromChat: (workId: string, message: string) => Promise<void>
  getRecommendations: (workId: string, nodeId?: string) => Promise<Idea[]>
}

export const useIdeaStore = create<IdeaState>()(
  immer((set, get) => ({
    ideas: [],
    selectedIdeaId: null,
    filter: {},
    isLoading: false,
    
    async loadIdeas(workId) {
      set({ isLoading: true })
      const ideas = await db.ideas.where('workId').equals(workId).toArray()
      set({ ideas, isLoading: false })
    },
    
    async createIdea(workId, data) {
      const idea: Idea = {
        id: generateId(),
        workId,
        ...data,
        status: 'pending',
        source: 'manual',
        createdAt: Date.now(),
        updatedAt: Date.now(),
      }
      await db.ideas.add(idea)
      set(state => { state.ideas.push(idea) })
      return idea.id
    },
    
    async markAsUsed(id, nodeId) {
      await db.ideas.update(id, {
        status: 'used',
        usedAt: Date.now(),
        usedInNodeId: nodeId,
      })
      set(state => {
        const idea = state.ideas.find(i => i.id === id)
        if (idea) {
          idea.status = 'used'
          idea.usedAt = Date.now()
          idea.usedInNodeId = nodeId
        }
      })
    },
    
    async extractFromChat(workId, message) {
      const extracted = await extractIdeasFromChat(workId, message)
      if (extracted.length > 0) {
        await db.ideas.bulkAdd(extracted)
        set(state => {
          state.ideas.push(...extracted)
        })
      }
    },
  }))
)
```
